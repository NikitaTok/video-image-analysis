def calculate_levels_params(self, peak_value):
    if peak_value <= 70:
        black = peak_value / 1.4
        white = peak_value * 1.84
        gamma = 1.35

        black = peak_value - 20
        white = peak_value * 1.8
        gamma = 1.43
    else:
        black = peak_value - 18
        white = peak_value * 1.84
